// Purpose: resolve benchmark profile from canonical business category / industry
// Output fields:
// - benchmark_profile
// - benchmark_meta
// - benchmark_confidence
//
// Notes:
// 1. This node does NOT compute audit metrics.
// 2. This node does NOT overwrite canonical business fields.
// 3. Downstream "Compute Audit Metrics" must read benchmarks ONLY from this node output.
// 4. All benchmark values are normalized numeric scales:
//    - rates: decimals (0.35 = 35%)
//    - roas: multiplier (3.0)
//    - cpl: currency/unit amount
//    - response_time_minutes: integer minutes

function pickFirst(obj, keys, fallback = null) {
  for (const key of keys) {
    if (obj[key] !== undefined && obj[key] !== null && obj[key] !== '') {
      return obj[key];
    }
  }
  return fallback;
}

function normalizeString(v) {
  if (v === null || v === undefined) return null;
  return String(v).trim().toLowerCase();
}

function resolveIndustryKey(input) {
  // Only canonical / likely canonical inputs.
  // Add/remove aliases here ONLY if still needed before schema is fully stabilized.
  const rawIndustry = pickFirst(input, [
    'industry',
    'industry_resolved',
    'business_category',
    'business_type',
    'niche'
  ]);

  const industry = normalizeString(rawIndustry);

  if (!industry) {
    return {
      key: 'generic',
      confidence: 'low',
      reason: 'missing_industry'
    };
  }

  // Local services cluster — includes exact form value 'local'
  if (
    industry === 'local' ||
    industry.includes('local service') ||
    industry.includes('local_service') ||
    industry.includes('home service') ||
    industry.includes('home_service') ||
    industry.includes('home_services') ||
    industry.includes('roofing') ||
    industry.includes('hvac') ||
    industry.includes('plumbing') ||
    industry.includes('electrical') ||
    industry.includes('contractor') ||
    industry.includes('garage door') ||
    industry.includes('landscaping') ||
    industry.includes('cleaning')
  ) {
    return {
      key: 'local_services',
      confidence: 'medium',
      reason: 'industry_keyword_match'
    };
  }

  // Agency / service businesses — includes exact form value 'agency'
  if (
    industry === 'agency' ||
    industry.includes('agency') ||
    industry.includes('marketing agency') ||
    industry.includes('lead gen agency') ||
    industry.includes('consulting') ||
    industry.includes('service business')
  ) {
    return {
      key: 'agency_services',
      confidence: 'medium',
      reason: 'industry_keyword_match'
    };
  }

  // Coaching / info — includes exact form value 'coaching'
  if (
    industry === 'coaching' ||
    industry.includes('coach') ||
    industry.includes('coaching') ||
    industry.includes('consultant') ||
    industry.includes('info product') ||
    industry.includes('course')
  ) {
    return {
      key: 'coaching_consulting',
      confidence: 'medium',
      reason: 'industry_keyword_match'
    };
  }

  // Clinics / medspa / dental — includes exact form value 'clinic'
  if (
    industry === 'clinic' ||
    industry.includes('clinic') ||
    industry.includes('medspa') ||
    industry.includes('dental') ||
    industry.includes('healthcare') ||
    industry.includes('chiropractic') ||
    industry.includes('medical')
  ) {
    return {
      key: 'clinic',
      confidence: 'medium',
      reason: 'industry_keyword_match'
    };
  }

  return {
    key: 'generic',
    confidence: 'low',
    reason: 'no_specific_match'
  };
}
