function firstDefined(...vals) {
  for (const v of vals) {
    if (v !== undefined && v !== null && v !== "") return v;
  }
  return undefined;
}

function pick(...keys) {
  for (const k of keys) {
    if (Object.prototype.hasOwnProperty.call(data, k)) {
      const v = data[k];
      if (v !== undefined && v !== null && v !== "") return v;
    }
  }
  return undefined;
}

function numOrNull(v) {
  if (v === undefined || v === null || v === "") return null;
  if (typeof v === "string") {
    const cleaned = v.replace(/[$,%\s,]/g, "");
    if (cleaned === "") return null;
    const n = Number(cleaned);
    return Number.isFinite(n) ? n : null;
  }
  const n = Number(v);
  return Number.isFinite(n) ? n : null;
}

function boolOrNull(v) {
  if (v === undefined || v === null || v === "") return null;
  if (typeof v === "boolean") return v;
  if (typeof v === "number") return v !== 0;
  if (typeof v === "string") {
    const s = v.trim().toLowerCase();
    if (["true", "yes", "1"].includes(s)) return true;
    if (["false", "no", "0"].includes(s)) return false;
  }
  return null;
}

function safeDivOrNull(a, b) {
  if (a === null || b === null) return null;
  if (!Number.isFinite(a) || !Number.isFinite(b) || b === 0) return null;
  return a / b;
}

function roundOrNull(v, digits = 2) {
  if (v === null || v === undefined || !Number.isFinite(v)) return null;
  const p = Math.pow(10, digits);
  return Math.round((v + Number.EPSILON) * p) / p;
}

function pctGapOrNull(actual, benchmark) {
  if (
    actual === null || actual === undefined ||
    benchmark === null || benchmark === undefined ||
    !Number.isFinite(actual) || !Number.isFinite(benchmark) ||
    benchmark === 0
  ) return null;
  return ((actual - benchmark) / benchmark) * 100;
}

function normalizeRate(v) {
  const n = numOrNull(v);
  if (n === null) return null;
  if (n > 1) return n / 100;
  if (n < 0) return null;
  return n;
}

function multiplyOrNull(...vals) {
  for (const v of vals) {
    if (v === null || v === undefined || !Number.isFinite(v)) return null;
  }
  return vals.reduce((acc, v) => acc * v, 1);
}

function subtractOrNull(a, b) {
  if (a === null || b === null || a === undefined || b === undefined) return null;
  if (!Number.isFinite(a) || !Number.isFinite(b)) return null;
  return a - b;
}

function addRisk(risks, key, condition) {
  if (condition) risks.push(key);
}

function statusForValue(v) {
  return v === null || v === undefined ? "insufficient_data" : "ok";
}

function buildStatusMap(obj) {
  const out = {};
  for (const [key, value] of Object.entries(obj)) {
    out[key] = statusForValue(value);
  }
  return out;
}

function hasBenchmarkValue(v) {
  return v !== null && v !== undefined && Number.isFinite(v);
}

function minMaxZero(v) {
  if (v === null || v === undefined || !Number.isFinite(v)) return null;
  return Math.max(0, v);
}

          improvement: roundOrNull(bmLeadToBooked - leadToBookedRate, 4),

          expected_additional_booked_calls: roundOrNull(additionalBooked, 2),
          expected_additional_completed_calls: roundOrNull(additionalCompleted, 2),
          expected_additional_closed_deals: roundOrNull(additionalClosed, 2),

          expected_additional_revenue: roundOrNull(additionalRevenue, 2),
          expected_additional_gross_profit: roundOrNull(additionalGrossProfit, 2)
        }
      };
    }

     // 2) Economics Efficiency
    else if (
      primaryBottleneck === "economics_efficiency" &&
      roas !== null &&
      bmRoas !== null &&
      adSpend !== null &&
      grossMargin !== null &&
      revenue !== null
    ) {
      const targetRevenue = adSpend * bmRoas;
      const targetGrossProfit = targetRevenue * grossMargin;

      const additionalRevenue = minMaxZero(targetRevenue - revenue);
      const additionalGrossProfit = minMaxZero(targetGrossProfit - (revenue * grossMargin));

      const additionalClosedDeals =
        finalAvgDealSize !== null && finalAvgDealSize > 0
          ? additionalRevenue / finalAvgDealSize
          : null;

      impactSimulation = {
        status: "ok",
        primary_fix: {
          metric: "roas",
          current: roundOrNull(roas, 4),
          target: roundOrNull(bmRoas, 4),
          improvement: roundOrNull(bmRoas - roas, 4),
          expected_additional_booked_calls: null,
          expected_additional_completed_calls: null,
          expected_additional_closed_deals: roundOrNull(additionalClosedDeals, 2),
          expected_additional_revenue: roundOrNull(additionalRevenue, 2),
          expected_additional_gross_profit: roundOrNull(additionalGrossProfit, 2),
          modeling_note: "Revenue uplift required to reach benchmark ROAS at current ad spend. Additional closed deals are estimated using current average deal size; booked and completed call counts are not inferred for economics-led simulations."
        }
      };
    }
    // 3) Booking -> Show
    else if (
      primaryBottleneck === "appointment_show_rate" &&
      bookedToCompletedRate !== null &&
      bmBookedToCompleted !== null &&
      callsBooked !== null &&
      completedToCloseRate !== null &&
      finalAvgDealSize !== null &&
      grossMargin !== null
    ) {
      const currentCompleted = callsCompleted ?? (callsBooked * bookedToCompletedRate);
      const targetCompleted = callsBooked * bmBookedToCompleted;

      const additionalCompleted = minMaxZero(targetCompleted - currentCompleted);
      const additionalClosed = additionalCompleted * completedToCloseRate;
      const additionalRevenue = additionalClosed * finalAvgDealSize;
      const additionalGrossProfit = additionalRevenue * grossMargin;

      impactSimulation = {
        status: "ok",
        primary_fix: {
          metric: "booked_to_completed_rate",
          current: roundOrNull(bookedToCompletedRate, 4),
          target: roundOrNull(bmBookedToCompleted, 4),
          improvement: roundOrNull(bmBookedToCompleted - bookedToCompletedRate, 4),

          expected_additional_booked_calls: 0,
          expected_additional_completed_calls: roundOrNull(additionalCompleted, 2),
          expected_additional_closed_deals: roundOrNull(additionalClosed, 2),

          expected_additional_revenue: roundOrNull(additionalRevenue, 2),
          expected_additional_gross_profit: roundOrNull(additionalGrossProfit, 2)
        }
      };
    }

    // 4) Show -> Close
    else if (
      primaryBottleneck === "sales_conversion" &&
      completedToCloseRate !== null &&
      bmCompletedToClose !== null &&
      callsCompleted !== null &&
      finalAvgDealSize !== null &&
      grossMargin !== null
    ) {
      const currentClosed = closedDeals ?? (callsCompleted * completedToCloseRate);
      const targetClosed = callsCompleted * bmCompletedToClose;

      const additionalClosed = minMaxZero(targetClosed - currentClosed);
      const additionalRevenue = additionalClosed * finalAvgDealSize;
      const additionalGrossProfit = additionalRevenue * grossMargin;

      impactSimulation = {
        status: "ok",
        primary_fix: {
          metric: "completed_to_close_rate",
          current: roundOrNull(completedToCloseRate, 4),
          target: roundOrNull(bmCompletedToClose, 4),
          improvement: roundOrNull(bmCompletedToClose - completedToCloseRate, 4),

          expected_additional_booked_calls: 0,
          expected_additional_completed_calls: 0,
          expected_additional_closed_deals: roundOrNull(additionalClosed, 2),

          expected_additional_revenue: roundOrNull(additionalRevenue, 2),
          expected_additional_gross_profit: roundOrNull(additionalGrossProfit, 2)
        }
      };
    }

    // 5) Economics efficiency (ROAS gap at current spend)
    else if (
      primaryBottleneck === "economics_efficiency" &&
      roas !== null &&
      bmRoas !== null &&
      adSpend !== null &&
      revenue !== null &&
      grossMargin !== null
    ) {
      const targetRevenue = adSpend * bmRoas;
      const additionalRevenue = minMaxZero(targetRevenue - revenue);
      const additionalGrossProfit = additionalRevenue * grossMargin;

      impactSimulation = {
        status: "ok",
        primary_fix: {
          metric: "roas",
          current: roundOrNull(roas, 4),
          target: roundOrNull(bmRoas, 4),
          improvement: roundOrNull(bmRoas - roas, 4),

          expected_additional_booked_calls: null,
          expected_additional_completed_calls: null,
          expected_additional_closed_deals: null,

          expected_additional_revenue: roundOrNull(additionalRevenue, 2),
          expected_additional_gross_profit: roundOrNull(additionalGrossProfit, 2),

          modeling_note: "Revenue uplift required to reach benchmark ROAS at current ad spend."
        }
      };
    }

    // 6) Speed to lead
    else if (primaryBottleneck === "speed_to_lead") {
      impactSimulation = {
        status: "not_modeled",
        primary_fix: {
          metric: "response_time_minutes",
          current: roundOrNull(responseTimeMinutes, 2),
          target: roundOrNull(bmResponseTime, 2),
          improvement: roundOrNull(
            responseTimeMinutes !== null && bmResponseTime !== null
              ? responseTimeMinutes - bmResponseTime
              : null,
            2
          )
        }
      };
    }
  } catch (e) {
    impactSimulation = {
      status: "error",
      error: String(e.message || e)
    };
  }
}

function isNum(v) {
  return typeof v === 'number' && Number.isFinite(v);
}

function clamp(v, min = 0, max = 100) {
  if (!isNum(v)) return null;
  return Math.max(min, Math.min(max, v));
}

function avg(values) {
  const valid = values.filter(isNum);
  if (!valid.length) return null;
  return valid.reduce((sum, v) => sum + v, 0) / valid.length;
}

function scoreAgainstBenchmark(actual, benchmark, mode = 'higher_better') {
  if (!isNum(actual) || !isNum(benchmark)) return null;
  if (benchmark <= 0) return null;
  if (actual < 0) return null;

  let raw = null;

  if (mode === 'higher_better') {
    raw = (actual / benchmark) * 100;
  } else if (mode === 'lower_better') {
    if (actual <= 0) return null;
    raw = (benchmark / actual) * 100;
  } else {
    return null;
  }

  return Math.round(clamp(raw, 0, 100)); // Using clamp to ensure 0-100 range
}

function metricValueIfOk(metricValue, fieldStatusLocal) {
  if (fieldStatusLocal !== undefined && fieldStatusLocal !== 'ok') return null;
  return isNum(metricValue) ? metricValue : null;
}
